component = code + design

binding:
	code -> design
		data binding:
			syntax: 
			a. interpolation: converts data into string
				code: title=''
				design: {{title}}
			b. property binding: retains original data type
				code: data=3000
				design: <element [attr]="data">


	design -> code
		event binding:
			design: <element (event)="M(args)">
			code: M(params){}
